EGA-Results 3D

This is the 3-d visualization for the EGA results.

MDS <- smacof::mds(dat_m, type = "ordinal", ndim = 3)
coordinates <- as.data.frame(MDS$conf) %>%
  tibble::rownames_to_column()

p <- plot_ly(coordinates,
                     x = ~D1,
                     y = ~D2,
                     z = ~D3) %>%
  add_markers(name = ~rowname,
              opacity = .80)
  
p
