library(ggplot2)
library(ggExtra)
p <- ggplot(gapminder::gapminder) +
     aes(x = gdpPercap, y = lifeExp, colour = continent, group = 1) +
     geom_smooth(method = "lm") +
     geom_point(alpha = .50) +
